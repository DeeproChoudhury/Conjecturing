lemma sqrt_nonnegative:
  fixes x::real
  shows "(x + 1)^2 >= 0"
proof -
  show ?thesis
  \<proof>
qed

lemma negative_square:
  fixes x::real
  assumes "x < 0"
  shows "x^2 < 0"
proof -
  show ?thesis
  \<proof>
qed

theorem real_number_inequality:
  fixes x::real
  shows "x^2 + 2*x + 1 >= 0"
proof -
  have c1 : "(x + 1)^2 = x^2 + 2*x + 1"
    by simp
  have c2 : "(x + 1)^2 >= 0"
    using sqrt_nonnegative
    sledgehammer
  then show ?thesis
    using c1
    sledgehammer
qed
