lemma nonneg_square:
  fixes x::real
  shows "x^2 >= 0"
proof -
  show ?thesis
  \<proof>
qed

lemma nonneg_square_plus_one:
  fixes x::real
  shows "(x + 1)^2 >= 0"
proof -
  show ?thesis
    using nonneg_square
    sledgehammer
qed

theorem real_number_inequality:
  fixes x::real
  shows "x^2 + 2*x + 1 >= 0"
proof -
  have "(x + 1)^2 >= 0"
    using nonneg_square_plus_one
    sledgehammer
  then show "x^2 + 2*x + 1 >= 0"
    using nonneg_square_plus_one
    sledgehammer
qed
